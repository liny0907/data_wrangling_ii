``` r
swm_html = 
  read_html("https://www.imdb.com/list/ls070150896/")

title_vec = 
  swm_html %>%
  html_elements(".lister-item-header a") %>%
  html_text()
gross_rev_vec = 
  swm_html %>%
  html_elements(".text-muted .ghost~ .text-muted+ span") %>%
  html_text()
```
